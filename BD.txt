CREATE PROCEDURE modificarTrabajo
@pais varchar(50), 
@region varchar(50), 
@ciudad varchar(60), 
@oficio varchar(30), 
@descripcion varchar(500), 
@IDtrabajo varchar(62) 
as 
UPDATE tblTrabajos 
SET pais=@pais,
region=@region,
ciudad=@ciudad,
oficio=@oficio,
descripcion=@descripcion 
where IDtrabajo=@IDtrabajo
